
# List all branches
git branch -a 

# Create a branch 
git branch "branch name"

# Switch to another branch 
git switch "branch name" 
    # May not work for older versions
git checkout "branch name"

# Merge branches
git merge "branch name"

# Delete a branch 
git branch -d "branch name"
